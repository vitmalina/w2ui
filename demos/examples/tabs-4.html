Add/Remove Tas

You can add remove tabs quickly or with animation.


$(function () { $('#tabs').w2tabs({ name: 'tabs', active: 'first', tabs: [ { id: 'first', caption: 'First Tab' } ] }); }); var ind = 1; function addTab() { if (ind > 12) return; w2ui.tabs.add({ id: 'tab'+ind, caption: 'Tab '+ ind }); ind++; } function addTabs() { if (ind > 9) return; w2ui.tabs.add([ { id: 'tab'+ind, caption: 'Tab '+ ind }, { id: 'tab'+(ind+1), caption: 'Tab '+ (ind+1) }, { id: 'tab'+(ind+2), caption: 'Tab '+ (ind+2) } ]); ind += 3; } function insertTab() { if (ind > 12) return; w2ui.tabs.animateInsert('first', { id: 'tab'+ ind, caption: 'Tab '+ ind }); ind++; } function removeTab() { if (ind == 1) return; ind--; w2ui.tabs.animateClose('tab'+ ind); } function removeTabs() { ind = 1; w2ui.tabs.remove('tab1', 'tab2', 'tab3', 'tab4', 'tab5', 'tab6', 'tab7', 'tab8', 'tab9', 'tab10', 'tab11', 'tab12'); }
